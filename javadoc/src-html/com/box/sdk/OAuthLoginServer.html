001package com.box.sdk;
002
003import java.io.IOException;
004import java.net.InetAddress;
005import java.net.ServerSocket;
006import java.security.SecureRandom;
007
008public class OAuthLoginServer {
009    private static final char[] HEX_ARRAY = "0123456789ABCDEF".toCharArray();
010    private static final int MIN_PORT = 49152;
011    private static final int MAX_PORT = 65535;
012    private static final int BACKLOG = 8;
013    private static final int STATE_LENGTH = 20;
014
015    private final ServerSocket socket;
016    private final int port;
017    private final String clientID;
018    private final String state;
019
020    public OAuthLoginServer(String clientID) throws IOException {
021        this.clientID = clientID;
022
023        SecureRandom random = new SecureRandom();
024        byte[] buffer = new byte[STATE_LENGTH];
025        random.nextBytes(buffer);
026        this.state = bytesToHex(buffer);
027
028        InetAddress address = InetAddress.getLocalHost();
029        ServerSocket socket = null;
030        int port;
031        for (port = MIN_PORT; port < MAX_PORT; ++port) {
032            try {
033                socket = new ServerSocket(port, BACKLOG, address);
034                port = port;
035                break;
036            } catch (IOException e) {
037                continue;
038            }
039        }
040
041        this.socket = socket;
042        this.port = port;
043    }
044
045    public String getRedirectURI() {
046        return String.format("https://localhost:{0}", this.port);
047    }
048
049    public String getLoginURL() {
050        return String.format("https://app.box.com/api/oauth2/authorize?response_type=code&client_id={0}&state={1}",
051            this.clientID, this.state);
052    }
053
054    public void listen() throws IOException {
055        this.socket.accept();
056    }
057
058    public void stop() throws IOException {
059        this.socket.close();
060    }
061
062    private static String bytesToHex(byte[] bytes) {
063        char[] hexChars = new char[bytes.length * 2];
064        for (int i = 0; i < bytes.length; ++i) {
065            int v = bytes[i] & 0xFF;
066            hexChars[i * 2] = HEX_ARRAY[v >>> 4];
067            hexChars[i * 2 + 1] = HEX_ARRAY[v & 0x0F];
068        }
069        return new String(hexChars);
070    }
071}
