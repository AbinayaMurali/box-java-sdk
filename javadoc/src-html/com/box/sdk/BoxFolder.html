013/**
014 * Represents a folder on Box. This class can be used to iterate through a folder's contents, collaborate a folder with
015 * another user or group, and perform other common folder operations (move, copy, delete, etc.).
016 */
017public final class BoxFolder extends BoxItem implements Iterable<BoxItem> {
018    private static final String UPLOAD_FILE_URL_BASE = "https://upload.box.com/api/2.0/";
019    private static final URLTemplate CREATE_FOLDER_URL = new URLTemplate("folders");
020    private static final URLTemplate COPY_FOLDER_URL = new URLTemplate("folders/%s/copy");
021    private static final URLTemplate DELETE_FOLDER_URL = new URLTemplate("folders/%s?recursive=%b");
022    private static final URLTemplate FOLDER_INFO_URL_TEMPLATE = new URLTemplate("folders/%s");
023    private static final URLTemplate UPLOAD_FILE_URL = new URLTemplate("files/content");
024    private static final URLTemplate ADD_COLLABORATION_URL = new URLTemplate("collaborations");
025    private static final URLTemplate GET_COLLABORATIONS_URL = new URLTemplate("folders/%s/collaborations");
026
027    private final URL folderURL;
028
029    /**
030     * Constructs a BoxFolder for a folder with a given ID.
031     * @param  api the API connection to be used by the folder.
032     * @param  id  the ID of the folder.
033     */
034    public BoxFolder(BoxAPIConnection api, String id) {
035        super(api, id);
036
037        this.folderURL = FOLDER_INFO_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
038    }
039
040    /**
041     * Gets the current user's root folder.
042     * @param  api the API connection to be used by the folder.
043     * @return     the user's root folder.
044     */
045    public static BoxFolder getRootFolder(BoxAPIConnection api) {
046        return new BoxFolder(api, "0");
047    }

049    /**
050     * Adds a collaborator to this folder.
051     * @param  collaborator the collaborator to add.
052     * @param  role         the role of the collaborator.
053     * @return              info about the new collaboration.
054     */
055    public BoxCollaboration.Info collaborate(BoxCollaborator collaborator, BoxCollaboration.Role role) {
056        JsonObject accessibleByField = new JsonObject();
057        accessibleByField.add("id", collaborator.getID());
058
059        if (collaborator instanceof BoxUser) {
060            accessibleByField.add("type", "user");
061        } else {
062            throw new IllegalArgumentException("The given collaborator is of an unknown type.");
063        }
064
065        return this.collaborate(accessibleByField, role);
066    }
067
068    /**
069     * Adds a collaborator to this folder. An email will be sent to the collaborator if they don't already have a Box
070     * account.
071     * @param  email the email address of the collaborator to add.
072     * @param  role  the role of the collaborator.
073     * @return       info about the new collaboration.
074     */
075    public BoxCollaboration.Info collaborate(String email, BoxCollaboration.Role role) {
076        JsonObject accessibleByField = new JsonObject();
077        accessibleByField.add("login", email);
078        accessibleByField.add("type", "user");
079
080        return this.collaborate(accessibleByField, role);
081    }
082
083    private BoxCollaboration.Info collaborate(JsonObject accessibleByField, BoxCollaboration.Role role) {
084        BoxAPIConnection api = this.getAPI();
085        URL url = ADD_COLLABORATION_URL.build(api.getBaseURL());
086
087        JsonObject itemField = new JsonObject();
088        itemField.add("id", this.getID());
089        itemField.add("type", "folder");
090
091        JsonObject requestJSON = new JsonObject();
092        requestJSON.add("item", itemField);
093        requestJSON.add("accessible_by", accessibleByField);
094        requestJSON.add("role", role.toJSONString());
095
096        BoxJSONRequest request = new BoxJSONRequest(api, url, "POST");
097        request.setBody(requestJSON.toString());
098        BoxJSONResponse response = (BoxJSONResponse) request.send();
099        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
100
101        BoxCollaboration newCollaboration = new BoxCollaboration(api, responseJSON.get("id").asString());
102        BoxCollaboration.Info info = newCollaboration.new Info(responseJSON);
103        return info;
104    }

106    /**
107     * Gets information about all of the collaborations for this folder.
108     * @return a collection of information about the collaborations for this folder.
109     */
110    public Collection<BoxCollaboration.Info> getCollaborations() {
111        BoxAPIConnection api = this.getAPI();
112        URL url = GET_COLLABORATIONS_URL.build(api.getBaseURL(), this.getID());
113
114        BoxAPIRequest request = new BoxAPIRequest(api, url, "GET");
115        BoxJSONResponse response = (BoxJSONResponse) request.send();
116        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
117
118        int entriesCount = responseJSON.get("total_count").asInt();
119        Collection<BoxCollaboration.Info> collaborations = new ArrayList<BoxCollaboration.Info>(entriesCount);
120        JsonArray entries = responseJSON.get("entries").asArray();
121        for (JsonValue entry : entries) {
122            JsonObject entryObject = entry.asObject();
123            BoxCollaboration collaboration = new BoxCollaboration(api, entryObject.get("id").asString());
124            BoxCollaboration.Info info = collaboration.new Info(entryObject);
125            collaborations.add(info);
126        }
127
128        return collaborations;
129    }

131    @Override
132    public BoxFolder.Info getInfo() {
133        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.folderURL, "GET");
134        BoxJSONResponse response = (BoxJSONResponse) request.send();
135        return new Info(response.getJSON());
136    }

138    @Override
139    public BoxFolder.Info getInfo(String... fields) {
140        String queryString = new QueryStringBuilder().addFieldsParam(fields).toString();
141        URL url = FOLDER_INFO_URL_TEMPLATE.buildWithQuery(this.getAPI().getBaseURL(), queryString, this.getID());
142
143        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
144        BoxJSONResponse response = (BoxJSONResponse) request.send();
145        return new Info(response.getJSON());
146    }

148    /**
149     * Updates the information about this folder with any info fields that have been modified locally.
150     * @param info the updated info.
151     */
152    public void updateInfo(BoxFolder.Info info) {
153        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), this.folderURL, "PUT");
154        request.setBody(info.getPendingChanges());
155        BoxJSONResponse response = (BoxJSONResponse) request.send();
156        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
157        info.update(jsonObject);
158    }

160    @Override
161    public BoxFolder.Info copy(BoxFolder destination) {
162        return this.copy(destination, null);
163    }
164
165    @Override
166    public BoxFolder.Info copy(BoxFolder destination, String newName) {
167        URL url = COPY_FOLDER_URL.build(this.getAPI().getBaseURL(), this.getID());
168        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
169
170        JsonObject parent = new JsonObject();
171        parent.add("id", destination.getID());
172
173        JsonObject copyInfo = new JsonObject();
174        copyInfo.add("parent", parent);
175        if (newName != null) {
176            copyInfo.add("name", newName);
177        }
178
179        request.setBody(copyInfo.toString());
180        BoxJSONResponse response = (BoxJSONResponse) request.send();
181        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
182        BoxFolder copiedFolder = new BoxFolder(this.getAPI(), responseJSON.get("id").asString());
183        return copiedFolder.new Info(responseJSON);
184    }

186    /**
187     * Creates a new child folder inside this folder.
188     * @param  name the new folder's name.
189     * @return      the created folder.
190     */
191    public BoxFolder createFolder(String name) {
192        JsonObject parent = new JsonObject();
193        parent.add("id", this.getID());
194
195        JsonObject newFolder = new JsonObject();
196        newFolder.add("name", name);
197        newFolder.add("parent", parent);
198
199        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), CREATE_FOLDER_URL.build(this.getAPI().getBaseURL()),
200            "POST");
201        request.setBody(newFolder.toString());
202        BoxJSONResponse response = (BoxJSONResponse) request.send();
203        JsonObject createdFolder = JsonObject.readFrom(response.getJSON());
204
205        return new BoxFolder(this.getAPI(), createdFolder.get("id").asString());
206    }

208    /**
209     * Deletes this folder, optionally recursively deleting all of its contents.
210     * @param recursive true to recursively delete this folder's contents; otherwise false.
211     */
212    public void delete(boolean recursive) {
213        URL url = DELETE_FOLDER_URL.build(this.getAPI().getBaseURL(), this.getID(), recursive);
214        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "DELETE");
215        BoxAPIResponse response = request.send();
216        response.disconnect();
217    }

219    /**
220     * Moves this folder to another folder.
221     * @param destination the destination folder.
222     */
223    public void move(BoxFolder destination) {
224        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), this.folderURL, "PUT");
225
226        JsonObject parent = new JsonObject();
227        parent.add("id", destination.getID());
228
229        JsonObject updateInfo = new JsonObject();
230        updateInfo.add("parent", parent);
231
232        request.setBody(updateInfo.toString());
233        BoxAPIResponse response = request.send();
234        response.disconnect();
235    }

246
247        request.setBody(updateInfo.toString());
248        BoxAPIResponse response = request.send();

277            .setSize(fileSize)
278            .setProgressListener(listener);

282    /**
283     * Uploads a new file to this folder with custom upload parameters.
284     * @param  uploadParams the custom upload parameters.
285     * @return              the uploaded file.
286     */
287    public BoxFile uploadFile(FileUploadParams uploadParams) {
288        URL uploadURL = UPLOAD_FILE_URL.build(UPLOAD_FILE_URL_BASE);
289        BoxMultipartRequest request = new BoxMultipartRequest(getAPI(), uploadURL);
290        request.putField("parent_id", getID());
291
292        if (uploadParams.getSize() > 0) {
293            request.setFile(uploadParams.getContent(), uploadParams.getName(), uploadParams.getSize());
294        } else {
295            request.setFile(uploadParams.getContent(), uploadParams.getName());
296        }
297
298        if (uploadParams.getCreated() != null) {
299            request.putField("content_created_at", uploadParams.getCreated());
300        }
301
302        if (uploadParams.getModified() != null) {
303            request.putField("content_modified_at", uploadParams.getModified());
304        }
305
306        BoxJSONResponse response;
307        if (uploadParams.getProgressListener() == null) {
308            response = (BoxJSONResponse) request.send();
309        } else {
310            response = (BoxJSONResponse) request.send(uploadParams.getProgressListener());
311        }
312        JsonObject collection = JsonObject.readFrom(response.getJSON());
313        JsonArray entries = collection.get("entries").asArray();
314        String uploadedFileID = entries.get(0).asObject().get("id").asString();
315
316        return new BoxFile(getAPI(), uploadedFileID);
317    }
