001package com.box.sdk;
002
003import java.util.List;
004
005import com.eclipsesource.json.JsonObject;
006
007public abstract class BoxResource {
008    private final BoxAPIConnection api;
009    private final String id;
010
011    public BoxResource(BoxAPIConnection api, String id) {
012        this.api = api;
013        this.id = id;
014    }
015
016    public BoxAPIConnection getAPI() {
017        return this.api;
018    }
019
020    public String getID() {
021        return this.id;
022    }
023
024    @Override
025    public boolean equals(Object other) {
026        if (other == null) {
027            return false;
028        }
029
030        if (other instanceof BoxResource) {
031            BoxResource otherResource = (BoxResource) other;
032            return this.getID().equals(otherResource.getID());
033        }
034
035        return false;
036    }
037
038    @Override
039    public int hashCode() {
040        return this.getID().hashCode();
041    }
042
043    public abstract class Info<T extends BoxResource> {
044        private JsonObject pendingChanges;
045
046        public Info() {
047            this.pendingChanges = new JsonObject();
048        }
049
050        public Info(String json) {
051            this(JsonObject.readFrom(json));
052        }
053
054        protected Info(JsonObject jsonObject) {
055            this.updateFromJSON(jsonObject);
056        }
057
058        public String getID() {
059            return BoxResource.this.getID();
060        }
061
062        public List<String> getPendingChanges() {
063            return this.pendingChanges.names();
064        }
065
066        public abstract T getResource();
067
068        @Override
069        public String toString() {
070            return this.pendingChanges.toString();
071        }
072
073        protected void addPendingChange(String key, String value) {
074            this.pendingChanges.set(key, value);
075        }
076
077        protected void clearPendingChanges() {
078            this.pendingChanges = new JsonObject();
079        }
080
081        protected void updateFromJSON(JsonObject jsonObject) {
082            for (JsonObject.Member member : jsonObject) {
083                if (member.getValue().isNull()) {
084                    continue;
085                }
086
087                this.parseJSONMember(member);
088            }
089
090            this.clearPendingChanges();
091        }
092
093        protected void parseJSONMember(JsonObject.Member member) { }
094    }
095}
