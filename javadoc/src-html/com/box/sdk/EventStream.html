001package com.box.sdk;
002
003import java.util.ArrayList;
004import java.util.Collection;
005import java.util.LinkedHashSet;
006
007import com.eclipsesource.json.JsonArray;
008import com.eclipsesource.json.JsonObject;
009import com.eclipsesource.json.JsonValue;
010
011public class EventStream {
012    private static final int LIMIT = 800;
013    private static final int LRU_SIZE = 512;
014    private static final URLTemplate EVENT_URL = new URLTemplate("events?limit=" + LIMIT + "&stream_position=%s");
015
016    private final BoxAPIConnection api;
017    private final Collection<EventListener> listeners;
018    private final Object listenerLock;
019    private final LinkedHashSet<String> receivedEvents;
020
021    private boolean started;
022    private Poller poller;
023    private Thread pollerThread;
024
025    public EventStream(BoxAPIConnection api) {
026        this.api = api;
027        this.listeners = new ArrayList<EventListener>();
028        this.listenerLock = new Object();
029        this.receivedEvents = new LinkedHashSet<String>(LRU_SIZE);
030    }
031
032    public void addListener(EventListener listener) {
033        synchronized (this.listenerLock) {
034            this.listeners.add(listener);
035        }
036    }
037
038    public boolean isStarted() {
039        return this.started;
040    }
041
042    public void stop() {
043        if (!this.started) {
044            throw new IllegalStateException("Cannot stop the EventStream because it isn't started.");
045        }
046
047        this.started = false;
048        this.pollerThread.interrupt();
049    }
050
051    public void start() {
052        if (this.started) {
053            throw new IllegalStateException("Cannot start the EventStream because it isn't stopped.");
054        }
055
056        BoxAPIRequest request = new BoxAPIRequest(this.api, EVENT_URL.build(this.api.getBaseURL(), "now"), "GET");
057        BoxJSONResponse response = (BoxJSONResponse) request.send();
058        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
059        final long initialPosition = jsonObject.get("next_stream_position").asLong();
060        this.poller = new Poller(initialPosition);
061
062        this.pollerThread = new Thread(this.poller);
063        this.pollerThread.setUncaughtExceptionHandler(new Thread.UncaughtExceptionHandler() {
064            public void uncaughtException(Thread t, Throwable e) {
065                EventStream.this.notifyException(e);
066            }
067        });
068        this.pollerThread.start();
069
070        this.started = true;
071    }
072
073    private void notifyEvent(BoxEvent event) {
074        synchronized (this.listenerLock) {
075            boolean newEvent = this.receivedEvents.add(event.getID());
076            if (newEvent) {
077                if (this.receivedEvents.size() > LRU_SIZE) {
078                    this.receivedEvents.iterator().remove();
079                }
080
081                for (EventListener listener : this.listeners) {
082                    listener.onEvent(event);
083                }
084            }
085        }
086    }
087
088    private void notifyException(Throwable e) {
089        if (e instanceof InterruptedException && !this.started) {
090            return;
091        }
092
093        this.stop();
094        synchronized (this.listenerLock) {
095            for (EventListener listener : this.listeners) {
096                if (listener.onException(e)) {
097                    return;
098                }
099            }
100        }
101    }
102
103    private class Poller implements Runnable {
104        private final long initialPosition;
105
106        private RealtimeServerConnection server;
107
108        public Poller(long initialPosition) {
109            this.initialPosition = initialPosition;
110            this.server = new RealtimeServerConnection(EventStream.this.api);
111        }
112
113        @Override
114        public void run() {
115            long position = this.initialPosition;
116            while (!Thread.interrupted()) {
117                if (this.server.getRemainingRetries() == 0) {
118                    this.server = new RealtimeServerConnection(EventStream.this.api);
119                }
120
121                if (this.server.waitForChange(position)) {
122                    if (Thread.interrupted()) {
123                        return;
124                    }
125
126                    BoxAPIRequest request = new BoxAPIRequest(EventStream.this.api,
127                        EVENT_URL.build(EventStream.this.api.getBaseURL(), position), "GET");
128                    BoxJSONResponse response = (BoxJSONResponse) request.send();
129                    JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
130                    position = jsonObject.get("next_stream_position").asLong();
131                    JsonArray entriesArray = jsonObject.get("entries").asArray();
132                    for (JsonValue entry : entriesArray) {
133                        BoxEvent event = new BoxEvent(EventStream.this.api, entry.asObject());
134                        EventStream.this.notifyEvent(event);
135                    }
136                }
137            }
138        }
139    }
140}
