003import java.io.IOException;
004import java.io.InputStream;

006import java.net.URL;

008import java.util.Collection;

015/**
016 * Represents an individual file on Box. This class can be used to download a file's contents, upload new versions, and
017 * perform other common file operations (move, copy, delete, etc.).
018 */
019public class BoxFile extends BoxItem {
020    private static final URLTemplate FILE_URL_TEMPLATE = new URLTemplate("files/%s");
021    private static final URLTemplate CONTENT_URL_TEMPLATE = new URLTemplate("files/%s/content");
022    private static final URLTemplate VERSIONS_URL_TEMPLATE = new URLTemplate("files/%s/versions");
023    private static final URLTemplate COPY_URL_TEMPLATE = new URLTemplate("files/%s/copy");
024    private static final int BUFFER_SIZE = 8192;
025
026    private final URL fileURL;
027    private final URL contentURL;
028
029    /**
030     * Constructs a BoxFile for a file with a given ID.
031     * @param  api the API connection to be used by the file.
032     * @param  id  the ID of the file.
033     */
034    public BoxFile(BoxAPIConnection api, String id) {
035        super(api, id);
036
037        this.fileURL = FILE_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
038        this.contentURL = CONTENT_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
039    }
040
041    /**
042     * Creates a new shared link for this file.
043     *
044     * <p>This method is a convenience method for manually creating a new shared link and applying it to this file with
045     * {@link Info#setSharedLink}. You may want to create the shared link manually so that it can be updated along with
046     * other changes to the file's info in a single network request, giving a boost to performance.</p>
047     *
048     * @param  access      the access level of the shared link.
049     * @param  unshareDate the date and time at which the link will expire. Can be null to create a non-expiring link.
050     * @param  permissions the permissions of the shared link. Can be null to use the default permissions.
051     * @return             the created shared link.
052     */
053    public BoxSharedLink createSharedLink(BoxSharedLink.Access access, Date unshareDate,
054        BoxSharedLink.Permissions permissions) {
055
056        BoxSharedLink sharedLink = new BoxSharedLink();
057        sharedLink.setAccess(access);
058
059        if (unshareDate != null) {
060            sharedLink.setUnsharedDate(unshareDate);
061        }
062
063        sharedLink.setPermissions(permissions);
064
065        Info info = new Info();
066        info.setSharedLink(sharedLink);
067
068        this.updateInfo(info);
069        return info.getSharedLink();
070    }
071
072    /**
073     * Downloads the contents of this file to a given OutputStream.
074     * @param output the stream to where the file will be written.
075     */
076    public void download(OutputStream output) {
077        this.download(output, null);
078    }
079
080    /**
081     * Downloads the contents of this file to a given OutputStream while reporting the progress to a ProgressListener.
082     * @param output   the stream to where the file will be written.
083     * @param listener a listener for monitoring the download's progress.
084     */
085    public void download(OutputStream output, ProgressListener listener) {
086        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.contentURL, "GET");
087        BoxAPIResponse response = request.send();
088        InputStream input = response.getBody();
089        if (listener != null) {
090            input = new ProgressInputStream(input, listener, response.getContentLength());
091        }
092
093        byte[] buffer = new byte[BUFFER_SIZE];
094        try {
095            int n = input.read(buffer);
096            while (n != -1) {
097                output.write(buffer, 0, n);
098                n = input.read(buffer);
099            }
100        } catch (IOException e) {
101            throw new BoxAPIException("Couldn't connect to the Box API due to a network error.", e);
102        }
103
104        response.disconnect();
105    }

107    @Override
108    public BoxFile.Info copy(BoxFolder destination) {
109        return this.copy(destination, null);
110    }
111
112    @Override
113    public BoxFile.Info copy(BoxFolder destination, String newName) {
114        URL url = COPY_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
115
116        JsonObject parent = new JsonObject();
117        parent.add("id", destination.getID());
118
119        JsonObject copyInfo = new JsonObject();
120        copyInfo.add("parent", parent);
121        if (newName != null) {
122            copyInfo.add("name", newName);
123        }
124
125        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
126        request.setBody(copyInfo.toString());
127        BoxJSONResponse response = (BoxJSONResponse) request.send();
128        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
129        BoxFile copiedFile = new BoxFile(this.getAPI(), responseJSON.get("id").asString());
130        return copiedFile.new Info(responseJSON);
131    }
132
133    /**
134     * Deletes this file by moving it to the trash.
135     */
136    public void delete() {
137        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.fileURL, "DELETE");
138        BoxAPIResponse response = request.send();
139        response.disconnect();
140    }
141
142    @Override
143    public BoxFile.Info getInfo() {
144        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.fileURL, "GET");
145        BoxJSONResponse response = (BoxJSONResponse) request.send();
146        return new Info(response.getJSON());
147    }
148
149    @Override
150    public BoxFile.Info getInfo(String... fields) {
151        String queryString = new QueryStringBuilder().addFieldsParam(fields).toString();
152        URL url = FILE_URL_TEMPLATE.buildWithQuery(this.getAPI().getBaseURL(), queryString, this.getID());
153
154        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
155        BoxJSONResponse response = (BoxJSONResponse) request.send();
156        return new Info(response.getJSON());
157    }
158
159    /**
160     * Updates the information about this file with any info fields that have been modified locally.
161     *
162     * <p>The only fields that will be updated are the ones that have been modified locally. For example, the following
163     * code won't update any information (or even send a network request) since none of the info's fields were
164     * changed:</p>
165     *
166     * <pre>BoxFile file = new File(api, id);
167     *BoxFile.Info info = file.getInfo();
168     *file.updateInfo(info);</pre>
169     *
170     * @param info the updated info.
171     */
172    public void updateInfo(BoxFile.Info info) {
173        URL url = FILE_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
174        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "PUT");
175        request.setBody(info.getPendingChanges());
176        BoxJSONResponse response = (BoxJSONResponse) request.send();
177        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
178        info.update(jsonObject);
179    }

181    /**
182     * Gets any previous versions of this file. Note that only users with premium accounts will be able to retrieve
183     * previous versions of their files.
184     * @return a list of previous file versions.
185     */
186    public Collection<BoxFileVersion> getVersions() {
187        URL url = VERSIONS_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
188        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
189        BoxJSONResponse response = (BoxJSONResponse) request.send();
190
191        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
192        JsonArray entries = jsonObject.get("entries").asArray();
193        Collection<BoxFileVersion> versions = new ArrayList<BoxFileVersion>();
194        for (JsonValue entry : entries) {
195            versions.add(new BoxFileVersion(this.getAPI(), entry.asObject(), this.getID()));
196        }
197
198        return versions;
199    }

201    /**
202     * Uploads a new version of this file, replacing the current version. Note that only users with premium accounts
203     * will be able to view and recover previous versions of the file.
204     * @param fileContent a stream containing the new file contents.
205     */
206    public void uploadVersion(InputStream fileContent) {
207        this.uploadVersion(fileContent, null);
208    }
209
210    /**
211     * Uploads a new version of this file, replacing the current version. Note that only users with premium accounts
212     * will be able to view and recover previous versions of the file.
213     * @param fileContent a stream containing the new file contents.
214     * @param modified    the date that the new version was modified.
215     */
216    public void uploadVersion(InputStream fileContent, Date modified) {
217        this.uploadVersion(fileContent, modified, 0, null);
218    }
219
220    /**
221     * Uploads a new version of this file, replacing the current version, while reporting the progress to a
222     * ProgressListener. Note that only users with premium accounts will be able to view and recover previous versions
223     * of the file.
224     * @param fileContent a stream containing the new file contents.
225     * @param modified    the date that the new version was modified.
226     * @param fileSize    the size of the file used for determining the progress of the upload.
227     * @param listener    a listener for monitoring the upload's progress.
228     */
229    public void uploadVersion(InputStream fileContent, Date modified, long fileSize, ProgressListener listener) {
230        URL uploadURL = CONTENT_URL_TEMPLATE.build(this.getAPI().getBaseUploadURL(), this.getID());
231        BoxMultipartRequest request = new BoxMultipartRequest(getAPI(), uploadURL);
232        if (fileSize > 0) {
233            request.setFile(fileContent, "", fileSize);
234        } else {
235            request.setFile(fileContent, "");
236        }
237
238        if (modified != null) {
239            request.putField("content_modified_at", modified);
240        }
241
242        BoxAPIResponse response;
243        if (listener == null) {
244            response = request.send();
245        } else {
246            response = request.send(listener);
247        }
248        response.disconnect();
249    }

251    /**
252     * Contains additional information about a BoxFile.
253     */
254    public class Info extends BoxItem.Info {
255        private String sha1;
256
257        /**
258         * Constructs an empty Info object.
259         */
260        public Info() {
261            super();
262        }
263
264        /**
265         * Constructs an Info object by parsing information from a JSON string.
266         * @param  json the JSON string to parse.
267         */
268        public Info(String json) {
269            super(json);
270        }
271
272        /**
273         * Constructs an Info object using an already parsed JSON object.
274         * @param  jsonObject the parsed JSON object.
275         */
276        protected Info(JsonObject jsonObject) {
277            super(jsonObject);
278        }
279
280        @Override
281        public BoxFile getResource() {
282            return BoxFile.this;
283        }
284
285        /**
286         * Gets the SHA1 hash of the file.
287         * @return the SHA1 hash of the file.
288         */
289        public String getSha1() {
290            return this.sha1;
291        }
292
293        @Override
294        protected void parseJSONMember(JsonObject.Member member) {
295            super.parseJSONMember(member);
296
297            String memberName = member.getName();
298            switch (memberName) {
299                case "sha1":
300                    this.sha1 = member.getValue().asString();
301                    break;
302                default:
303                    break;
304            }
305        }
306    }
