001package com.box.sdk;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.URL;
007import java.util.ArrayList;
008import java.util.Collection;
009import java.util.Date;
010
011import com.eclipsesource.json.JsonArray;
012import com.eclipsesource.json.JsonObject;
013import com.eclipsesource.json.JsonValue;
014
015public class BoxFile extends BoxItem {
016    private static final URLTemplate FILE_URL_TEMPLATE = new URLTemplate("files/%s");
017    private static final URLTemplate CONTENT_URL_TEMPLATE = new URLTemplate("files/%s/content");
018    private static final URLTemplate VERSIONS_URL_TEMPLATE = new URLTemplate("files/%s/versions");
019    private static final URLTemplate COPY_URL_TEMPLATE = new URLTemplate("files/%s/copy");
020    private static final int BUFFER_SIZE = 8192;
021
022    private final URL fileURL;
023    private final URL contentURL;
024
025    public BoxFile(BoxAPIConnection api, String id) {
026        super(api, id);
027
028        this.fileURL = FILE_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
029        this.contentURL = CONTENT_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
030    }
031
032    public void download(OutputStream output) {
033        this.download(output, null);
034    }
035
036    public void download(OutputStream output, ProgressListener listener) {
037        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.contentURL, "GET");
038        BoxAPIResponse response = request.send();
039        InputStream input = response.getBody();
040
041        long totalRead = 0;
042        byte[] buffer = new byte[BUFFER_SIZE];
043        try {
044            int n = input.read(buffer);
045            totalRead += n;
046            while (n != -1) {
047                output.write(buffer, 0, n);
048                if (listener != null) {
049                    listener.onProgressChanged(totalRead, response.getContentLength());
050                }
051                n = input.read(buffer);
052                totalRead += n;
053            }
054        } catch (IOException e) {
055            throw new BoxAPIException("Couldn't connect to the Box API due to a network error.", e);
056        }
057
058        response.disconnect();
059    }
060
061    public BoxFile.Info copy(BoxFolder destination) {
062        return this.copy(destination, null);
063    }
064
065    public BoxFile.Info copy(BoxFolder destination, String newName) {
066        URL url = COPY_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
067
068        JsonObject parent = new JsonObject();
069        parent.add("id", destination.getID());
070
071        JsonObject copyInfo = new JsonObject();
072        copyInfo.add("parent", parent);
073        if (newName != null) {
074            copyInfo.add("name", newName);
075        }
076
077        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
078        request.setBody(copyInfo.toString());
079        BoxJSONResponse response = (BoxJSONResponse) request.send();
080        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
081        BoxFile copiedFile = new BoxFile(this.getAPI(), responseJSON.get("id").asString());
082        return copiedFile.new Info(responseJSON);
083    }
084
085    public void delete() {
086        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.fileURL, "DELETE");
087        BoxAPIResponse response = request.send();
088        response.disconnect();
089    }
090
091    public BoxFile.Info getInfo() {
092        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.fileURL, "GET");
093        BoxJSONResponse response = (BoxJSONResponse) request.send();
094        return new Info(response.getJSON());
095    }
096
097    public BoxFile.Info getInfo(String... fields) {
098        String queryString = new QueryStringBuilder().addFieldsParam(fields).toString();
099        URL url = FILE_URL_TEMPLATE.buildWithQuery(this.getAPI().getBaseURL(), queryString, this.getID());
100
101        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
102        BoxJSONResponse response = (BoxJSONResponse) request.send();
103        return new Info(response.getJSON());
104    }
105
106    public void updateInfo(BoxFile.Info info) {
107        URL url = FILE_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
108        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "PUT");
109        request.setBody(info.getPendingChanges());
110        BoxJSONResponse response = (BoxJSONResponse) request.send();
111        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
112        info.update(jsonObject);
113    }
114
115    public Collection<BoxFileVersion> getVersions() {
116        URL url = VERSIONS_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
117        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
118        BoxJSONResponse response = (BoxJSONResponse) request.send();
119
120        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
121        JsonArray entries = jsonObject.get("entries").asArray();
122        Collection<BoxFileVersion> versions = new ArrayList<BoxFileVersion>();
123        for (JsonValue entry : entries) {
124            versions.add(new BoxFileVersion(this.getAPI(), entry.asObject(), this.getID()));
125        }
126
127        return versions;
128    }
129
130    public void uploadVersion(InputStream fileContent) {
131        this.uploadVersion(fileContent, null);
132    }
133
134    public void uploadVersion(InputStream fileContent, Date modified) {
135        URL uploadURL = CONTENT_URL_TEMPLATE.build(this.getAPI().getBaseUploadURL(), this.getID());
136        BoxMultipartRequest request = new BoxMultipartRequest(getAPI(), uploadURL);
137        request.setFile(fileContent, "");
138
139        if (modified != null) {
140            request.putField("content_modified_at", modified);
141        }
142
143        BoxAPIResponse response = request.send();
144        response.disconnect();
145    }
146
147    public class Info extends BoxItem.Info<BoxFile> {
148        private String sha1;
149
150        public Info() {
151            super();
152        }
153
154        public Info(String json) {
155            super(json);
156        }
157
158        protected Info(JsonObject jsonObject) {
159            super(jsonObject);
160        }
161
162        @Override
163        public BoxFile getResource() {
164            return BoxFile.this;
165        }
166
167        public String getSha1() {
168            return this.sha1;
169        }
170
171        @Override
172        protected void parseJSONMember(JsonObject.Member member) {
173            super.parseJSONMember(member);
174
175            String memberName = member.getName();
176            switch (memberName) {
177                case "sha1":
178                    this.sha1 = member.getValue().asString();
179                    break;
180                default:
181                    break;
182            }
183        }
184    }
185}
