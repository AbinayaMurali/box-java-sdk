001package com.box.sdk;
002
003import java.io.InputStream;
004import java.net.URL;
005import java.util.Date;
006import java.util.Iterator;
007
008import com.eclipsesource.json.JsonArray;
009import com.eclipsesource.json.JsonObject;
010
011public final class BoxFolder extends BoxItem implements Iterable<BoxItem> {
012    private static final String UPLOAD_FILE_URL_BASE = "https://upload.box.com/api/2.0/";
013    private static final URLTemplate CREATE_FOLDER_URL = new URLTemplate("folders");
014    private static final URLTemplate COPY_FOLDER_URL = new URLTemplate("folders/%s/copy");
015    private static final URLTemplate DELETE_FOLDER_URL = new URLTemplate("folders/%s?recursive=%b");
016    private static final URLTemplate FOLDER_INFO_URL_TEMPLATE = new URLTemplate("folders/%s");
017    private static final URLTemplate UPLOAD_FILE_URL = new URLTemplate("files/content");
018
019    private final URL folderURL;
020
021    public BoxFolder(BoxAPIConnection api, String id) {
022        super(api, id);
023
024        this.folderURL = FOLDER_INFO_URL_TEMPLATE.build(this.getAPI().getBaseURL(), this.getID());
025    }
026
027    public static BoxFolder getRootFolder(BoxAPIConnection api) {
028        return new BoxFolder(api, "0");
029    }
030
031    public BoxFolder.Info getInfo() {
032        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), this.folderURL, "GET");
033        BoxJSONResponse response = (BoxJSONResponse) request.send();
034        return new Info(response.getJSON());
035    }
036
037    public BoxFolder.Info getInfo(String... fields) {
038        StringBuilder fieldsStringBuilder = new StringBuilder();
039        for (String field : fields) {
040            fieldsStringBuilder.append(field);
041            fieldsStringBuilder.append(",");
042        }
043        fieldsStringBuilder.deleteCharAt(fieldsStringBuilder.length() - 1);
044
045        String[][] queryParams = new String[][] {{"fields", fieldsStringBuilder.toString()}};
046        URL url = FOLDER_INFO_URL_TEMPLATE.build(this.getAPI().getBaseURL(), queryParams, this.getID());
047        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "GET");
048        BoxJSONResponse response = (BoxJSONResponse) request.send();
049        return new Info(response.getJSON());
050    }
051
052    public void updateInfo(BoxFolder.Info info) {
053        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), this.folderURL, "PUT");
054        request.setBody(info.toString());
055        BoxJSONResponse response = (BoxJSONResponse) request.send();
056        JsonObject jsonObject = JsonObject.readFrom(response.getJSON());
057        info.updateFromJSON(jsonObject);
058    }
059
060    public BoxFolder.Info copy(BoxFolder destination) {
061        return this.copy(destination, null);
062    }
063
064    public BoxFolder.Info copy(BoxFolder destination, String newName) {
065        return this.copy(destination.getID(), newName);
066    }
067
068    public BoxFolder.Info copy(String destinationID) {
069        return this.copy(destinationID, null);
070    }
071
072    public BoxFolder.Info copy(String destinationID, String newName) {
073        URL url = COPY_FOLDER_URL.build(this.getAPI().getBaseURL(), this.getID());
074        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), url, "POST");
075
076        JsonObject parent = new JsonObject();
077        parent.add("id", destinationID);
078
079        JsonObject copyInfo = new JsonObject();
080        copyInfo.add("parent", parent);
081        if (newName != null) {
082            copyInfo.add("name", newName);
083        }
084
085        request.setBody(copyInfo.toString());
086        BoxJSONResponse response = (BoxJSONResponse) request.send();
087        JsonObject responseJSON = JsonObject.readFrom(response.getJSON());
088        BoxFolder copiedFolder = new BoxFolder(this.getAPI(), responseJSON.get("id").asString());
089        return copiedFolder.new Info(responseJSON);
090    }
091
092    public BoxFolder createFolder(String name) {
093        JsonObject parent = new JsonObject();
094        parent.add("id", this.getID());
095
096        JsonObject newFolder = new JsonObject();
097        newFolder.add("name", name);
098        newFolder.add("parent", parent);
099
100        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), CREATE_FOLDER_URL.build(this.getAPI().getBaseURL()),
101            "POST");
102        request.setBody(newFolder.toString());
103        BoxJSONResponse response = (BoxJSONResponse) request.send();
104        JsonObject createdFolder = JsonObject.readFrom(response.getJSON());
105
106        return new BoxFolder(this.getAPI(), createdFolder.get("id").asString());
107    }
108
109    public void delete(boolean recursive) {
110        URL url = DELETE_FOLDER_URL.build(this.getAPI().getBaseURL(), this.getID(), recursive);
111        BoxAPIRequest request = new BoxAPIRequest(this.getAPI(), url, "DELETE");
112        BoxAPIResponse response = request.send();
113        response.disconnect();
114    }
115
116    public void move(BoxFolder destination) {
117        this.move(destination.getID());
118    }
119
120    public void move(String destinationID) {
121        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), this.folderURL, "PUT");
122
123        JsonObject parent = new JsonObject();
124        parent.add("id", destinationID);
125
126        JsonObject updateInfo = new JsonObject();
127        updateInfo.add("parent", parent);
128
129        request.setBody(updateInfo.toString());
130        BoxAPIResponse response = request.send();
131        response.disconnect();
132    }
133
134    public void rename(String newName) {
135        BoxJSONRequest request = new BoxJSONRequest(this.getAPI(), this.folderURL, "PUT");
136
137        JsonObject updateInfo = new JsonObject();
138        updateInfo.add("name", newName);
139
140        request.setBody(updateInfo.toString());
141        BoxAPIResponse response = request.send();
142        response.disconnect();
143    }
144
145    public BoxFile uploadFile(InputStream fileContent, String name, Date created, Date modified) {
146        URL uploadURL = UPLOAD_FILE_URL.build(UPLOAD_FILE_URL_BASE);
147        BoxMultipartRequest request = new BoxMultipartRequest(getAPI(), uploadURL);
148        request.putField("parent_id", getID());
149        request.setFile(fileContent, name);
150
151        BoxJSONResponse response = (BoxJSONResponse) request.send();
152        JsonObject collection = JsonObject.readFrom(response.getJSON());
153        JsonArray entries = collection.get("entries").asArray();
154        String uploadedFileID = entries.get(0).asObject().get("id").asString();
155
156        return new BoxFile(getAPI(), uploadedFileID);
157    }
158
159    public Iterator<BoxItem> iterator() {
160        return new BoxItemIterator(BoxFolder.this.getAPI(), BoxFolder.this.getID());
161    }
162

164        public Info() {
165            super();
166        }
167

169            super(json);
170        }
171
172        protected Info(JsonObject jsonObject) {
173            super(jsonObject);
174        }
175
176        @Override
177        public BoxFolder getResource() {
178            return BoxFolder.this;
179        }

181        @Override
182        protected void parseJSONMember(JsonObject.Member member) {
183            super.parseJSONMember(member);
184
185            String memberName = member.getName();
186            switch (memberName) {
187                default:
188                    break;
189            }
190        }
191    }
