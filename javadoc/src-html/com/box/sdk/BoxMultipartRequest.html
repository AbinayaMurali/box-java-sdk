001package com.box.sdk;
002
003import java.io.IOException;
004import java.io.InputStream;
005import java.io.OutputStream;
006import java.net.HttpURLConnection;
007import java.net.URL;
008import java.nio.charset.StandardCharsets;
009import java.util.Date;
010import java.util.HashMap;
011import java.util.Map;
012import java.util.logging.Level;
013import java.util.logging.Logger;
014
015public class BoxMultipartRequest extends BoxAPIRequest {
016    private static final Logger LOGGER = Logger.getLogger(BoxFolder.class.getName());
017    private static final String BOUNDARY = "da39a3ee5e6b4b0d3255bfef95601890afd80709";
018    private static final int BUFFER_SIZE = 8192;
019
020    private final StringBuilder loggedRequest = new StringBuilder();
021
022    private OutputStream outputStream;
023    private InputStream inputStream;
024    private String filename;
025    private long fileSize;
026    private Map<String, String> fields;
027    private boolean firstBoundary;
028
029    public BoxMultipartRequest(BoxAPIConnection api, URL url) {
030        super(api, url, "POST");
031
032        this.fields = new HashMap<String, String>();
033        this.firstBoundary = true;
034
035        this.addHeader("Content-Type", "multipart/form-data; boundary=" + BOUNDARY);
036    }
037
038    public void putField(String key, String value) {
039        this.fields.put(key, value);
040    }
041
042    public void putField(String key, Date value) {
043        this.fields.put(key, String.format("%FT%T-00:00", value));
044    }
045
046    public void setFile(InputStream inputStream, String filename) {
047        this.inputStream = inputStream;
048        this.filename = filename;
049    }
050
051    public void setFile(InputStream inputStream, String filename, long fileSize) {
052        this.setFile(inputStream, filename);
053        this.fileSize = fileSize;
054    }
055
056    @Override
057    public void setBody(InputStream stream) {
058        throw new UnsupportedOperationException();
059    }
060
061    @Override
062    public void setBody(String body) {
063        throw new UnsupportedOperationException();
064    }
065
066    @Override
067    public void writeBody(HttpURLConnection connection, ProgressListener listener) {
068        try {
069            connection.setChunkedStreamingMode(0);
070            connection.setDoOutput(true);
071            this.outputStream = connection.getOutputStream();
072
073            this.writePartHeader(new String[][] {{"name", "filename"}, {"filename", this.filename}},
074                "application/octet-stream");
075
076            OutputStream fileContentsOutputStream = this.outputStream;
077            if (listener != null) {
078                fileContentsOutputStream = new ProgressOutputStream(this.outputStream, listener, this.fileSize);
079            }
080            byte[] buffer = new byte[BUFFER_SIZE];
081            int n = this.inputStream.read(buffer);
082            while (n != -1) {
083                fileContentsOutputStream.write(buffer, 0, n);
084                n = this.inputStream.read(buffer);
085            }
086
087            if (LOGGER.isLoggable(Level.INFO)) {
088                this.loggedRequest.append("<File Contents Omitted>");
089            }
090
091            for (Map.Entry<String, String> entry : this.fields.entrySet()) {
092                this.writePartHeader(new String[][] {{"name", entry.getKey()}});
093                this.writeOutput(entry.getValue());
094            }
095
096            this.writeBoundary();
097        } catch (IOException e) {
098            throw new BoxAPIException("Couldn't connect to the Box API due to a network error.", e);
099        }
100    }
101
102    @Override
103    protected void resetBody() throws IOException {
104        this.firstBoundary = true;
105        this.inputStream.reset();
106        this.loggedRequest.setLength(0);
107    }
108
109    @Override
110    protected String bodyToString() {
111        return this.loggedRequest.toString();
112    }
113
114    private void writeBoundary() throws IOException {
115        if (!this.firstBoundary) {
116            this.writeOutput("\r\n");
117        }
118
119        this.firstBoundary = false;
120        this.writeOutput("--");
121        this.writeOutput(BOUNDARY);
122    }
123
124    private void writePartHeader(String[][] formData) throws IOException {
125        this.writePartHeader(formData, null);
126    }
127
128    private void writePartHeader(String[][] formData, String contentType) throws IOException {
129        this.writeBoundary();
130        this.writeOutput("\r\n");
131        this.writeOutput("Content-Disposition: form-data");
132        for (int i = 0; i < formData.length; i++) {
133            this.writeOutput("; ");
134            this.writeOutput(formData[i][0]);
135            this.writeOutput("=\"");
136            this.writeOutput(formData[i][1]);
137            this.writeOutput("\"");
138        }
139
140        if (contentType != null) {
141            this.writeOutput("\r\nContent-Type: ");
142            this.writeOutput(contentType);
143        }
144
145        this.writeOutput("\r\n\r\n");
146    }
147
148    private void writeOutput(String s) throws IOException {
149        this.outputStream.write(s.getBytes(StandardCharsets.UTF_8));
150        if (LOGGER.isLoggable(Level.INFO)) {
151            this.loggedRequest.append(s);
152        }
153    }
154
155    private void writeOutput(int b) throws IOException {
156        this.outputStream.write(b);
157    }
158}
