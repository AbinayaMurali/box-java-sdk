001package com.box.sdk;
002
003import java.text.DateFormat;
004import java.text.ParseException;
005import java.text.SimpleDateFormat;
006import java.util.Date;
007
008public final class BoxDateParser {
009    private static final ThreadLocal<DateFormat> THREAD_LOCAL_DATE_FORMAT = new ThreadLocal<DateFormat>() {
010        @Override
011        protected DateFormat initialValue() {
012            return new SimpleDateFormat("yyyy-MM-dd'T'HH:mm:ssX");
013        }
014    };
015
016    private BoxDateParser() { }
017
018    public static Date parse(String dateString) throws ParseException {
019        return THREAD_LOCAL_DATE_FORMAT.get().parse(dateString);
020    }
021}
