001package com.box.sdk;
002
003import com.eclipsesource.json.JsonObject;
004import com.eclipsesource.json.JsonValue;
005
006public class BoxEvent extends BoxResource {
007    private BoxResource source;
008    private BoxEvent.Type type;
009
010    public BoxEvent(BoxAPIConnection api, JsonObject jsonObject) {
011        super(api, jsonObject.get("event_id").asString());
012
013        for (JsonObject.Member member : jsonObject) {
014            if (member.getValue().isNull()) {
015                continue;
016            }
017
018            this.parseJsonMember(member);
019        }
020    }
021
022    public BoxResource getSource() {
023        return this.source;
024    }
025
026    public BoxEvent.Type getType() {
027        return this.type;
028    }
029
030    protected void parseJsonMember(JsonObject.Member member) {
031        JsonValue value = member.getValue();
032        if (value.isNull()) {
033            return;
034        }
035
036        switch (member.getName()) {
037            case "source":
038                this.source = this.parseSource(value.asObject());
039                break;
040            case "event_type":
041                this.type = BoxEvent.Type.valueOf(value.asString());
042                break;
043            default:
044                break;
045        }
046    }
047
048    private BoxResource parseSource(JsonObject jsonObject) {
049        String type = jsonObject.get("type").asString();
050        switch (type) {
051            case "folder":
052                return new BoxFolder(this.getAPI(), jsonObject.get("id").asString());
053            case "file":
054                return new BoxFile(this.getAPI(), jsonObject.get("id").asString());
055            default:
056                return null;
057        }
058    }
059
060    public enum Type {
061        ITEM_CREATE,
062        ITEM_UPLOAD,
063        COMMENT_CREATE,
064        ITEM_DOWNLOAD,
065        ITEM_PREVIEW,
066        ITEM_MOVE,
067        ITEM_COPY,
068        TASK_ASSIGNMENT_CREATE,
069        LOCK_CREATE,
070        LOCK_DESTROY,
071        ITEM_TRASH,
072        ITEM_UNDELETE_VIA_TRASH,
073        COLLAB_ADD_COLLABORATOR,
074        COLLAB_INVITE_COLLABORATOR,
075        ITEM_SYNC,
076        ITEM_UNSYNC,
077        ITEM_RENAME,
078        ITEM_SHARED_CREATE,
079        ITEM_SHARED_UNSHARE,
080        ITEM_SHARED,
081        TAG_ITEM_CREATE,
082        ADD_LOGIN_ACTIVITY_DEVICE,
083        REMOVE_LOGIN_ACTIVITY_DEVICE,
084        CHANGE_ADMIN_ROLE;
085    }
086}
