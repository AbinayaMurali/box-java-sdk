001package com.box.sdk;
002
003import java.text.ParseException;
004import java.util.ArrayList;
005import java.util.Date;
006import java.util.List;
007
008import com.eclipsesource.json.JsonArray;
009import com.eclipsesource.json.JsonObject;
010import com.eclipsesource.json.JsonValue;
011
012public abstract class BoxItem extends BoxResource {
013    public BoxItem(BoxAPIConnection api, String id) {
014        super(api, id);
015    }
016
017    public abstract class Info<T extends BoxItem> extends BoxResource.Info<T> {
018        private String sequenceID;
019        private String etag;
020        private String name;
021        private Date createdAt;
022        private Date modifiedAt;
023        private String description;
024        private long size;
025        private List<BoxFolder> pathCollection;
026        private BoxUser.Info createdBy;
027        private BoxUser.Info modifiedBy;
028        private Date trashedAt;
029        private Date purgedAt;
030        private Date contentCreatedAt;
031        private Date contentModifiedAt;
032        private BoxUser.Info ownedBy;
033        private List<String> tags;
034        private BoxFolder.Info parent;
035
036        public Info() {
037            super();
038        }
039
040        public Info(String json) {
041            super(json);
042        }
043
044        protected Info(JsonObject jsonObject) {
045            super(jsonObject);
046        }
047
048        public String getEtag() {
049            return this.etag;
050        }
051
052        public String getName() {
053            return this.name;
054        }
055
056        public void setName(String name) {
057            this.name = name;
058            this.addPendingChange("name", name);
059        }
060
061        public Date getCreatedAt() {
062            return this.createdAt;
063        }
064
065        public Date getModifiedAt() {
066            return this.modifiedAt;
067        }
068
069        public String getDescription() {
070            return this.description;
071        }
072
073        public void setDescription(String description) {
074            this.description = description;
075            this.addPendingChange("description", description);
076        }
077
078        public long getSize() {
079            return this.size;
080        }
081
082        public List<BoxFolder> getPathCollection() {
083            return this.pathCollection;
084        }
085
086        public BoxUser.Info getCreatedBy() {
087            return this.createdBy;
088        }
089
090        public BoxUser.Info getModifiedBy() {
091            return this.modifiedBy;
092        }
093
094        public Date getTrashedAt() {
095            return this.trashedAt;
096        }
097
098        public Date getPurgedAt() {
099            return this.purgedAt;
100        }
101
102        public Date getContentCreatedAt() {
103            return this.createdAt;
104        }
105
106        public Date getContentModifiedAt() {
107            return this.contentModifiedAt;
108        }
109
110        public BoxUser.Info getOwnedBy() {
111            return this.ownedBy;
112        }
113
114        public String getSequenceID() {
115            return this.sequenceID;
116        }
117
118        public List<String> getTags() {
119            return this.tags;
120        }
121
122        public BoxFolder.Info getParent() {
123            return this.parent;
124        }
125
126        @Override
127        protected void parseJSONMember(JsonObject.Member member) {
128            super.parseJSONMember(member);
129
130            try {
131                JsonValue value = member.getValue();
132                switch (member.getName()) {
133                    case "sequence_id":
134                        this.sequenceID = value.asString();
135                        break;
136                    case "etag":
137                        this.etag = value.asString();
138                        break;
139                    case "name":
140                        this.name = value.asString();
141                        break;
142                    case "created_at":
143                        this.createdAt = BoxDateParser.parse(value.asString());
144                        break;
145                    case "modified_at":
146                        this.modifiedAt = BoxDateParser.parse(value.asString());
147                        break;
148                    case "description":
149                        this.description = value.asString();
150                        break;
151                    case "size":
152                        this.size = Double.valueOf(value.toString()).longValue();
153                        break;
154                    case "trashed_at":
155                        this.trashedAt = BoxDateParser.parse(value.asString());
156                        break;
157                    case "purged_at":
158                        this.purgedAt = BoxDateParser.parse(value.asString());
159                        break;
160                    case "content_created_at":
161                        this.contentCreatedAt = BoxDateParser.parse(value.asString());
162                        break;
163                    case "content_modified_at":
164                        this.contentModifiedAt = BoxDateParser.parse(value.asString());
165                        break;
166                    case "path_collection":
167                        this.pathCollection = this.parsePathCollection(value.asObject());
168                        break;
169                    case "created_by":
170                        this.createdBy = this.parseUserInfo(value.asObject());
171                        break;
172                    case "modified_by":
173                        this.modifiedBy = this.parseUserInfo(value.asObject());
174                        break;
175                    case "owned_by":
176                        this.ownedBy = this.parseUserInfo(value.asObject());
177                        break;
178                    case "tags":
179                        this.tags = this.parseTags(value.asArray());
180                        break;
181                    case "parent":
182                        JsonObject jsonObject = value.asObject();
183                        String id = jsonObject.get("id").asString();
184                        BoxFolder parentFolder = new BoxFolder(getAPI(), id);
185                        this.parent = parentFolder.new Info(jsonObject);
186                        break;
187                    default:
188                        break;
189                }
190            } catch (ParseException e) {
191                assert false : "A ParseException indicates a bug in the SDK.";
192            }
193        }
194
195        private List<BoxFolder> parsePathCollection(JsonObject jsonObject) {
196            int count = jsonObject.get("total_count").asInt();
197            List<BoxFolder> pathCollection = new ArrayList<BoxFolder>(count);
198            JsonArray entries = jsonObject.get("entries").asArray();
199            for (JsonValue value : entries) {
200                JsonObject entry = value.asObject();
201                String id = entry.get("id").asString();
202                pathCollection.add(new BoxFolder(getAPI(), id));
203            }
204
205            return pathCollection;
206        }
207
208        private BoxUser.Info parseUserInfo(JsonObject jsonObject) {
209            String userID = jsonObject.get("id").asString();
210            BoxUser user = new BoxUser(getAPI(), userID);
211            return user.new Info(jsonObject);
212        }
213
214        private List<String> parseTags(JsonArray jsonArray) {
215            List<String> tags = new ArrayList<String>(jsonArray.size());
216            for (JsonValue value : jsonArray) {
217                tags.add(value.asString());
218            }
219
220            return tags;
221        }
222    }
223}
