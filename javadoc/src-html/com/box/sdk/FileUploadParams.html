001package com.box.sdk;
002
003import java.io.InputStream;
004import java.util.Date;
005
006public class FileUploadParams {
007    private InputStream content;
008    private String name;
009    private Date created;
010    private Date modified;
011    private long size;
012    private ProgressListener listener;
013
014    public InputStream getContent() {
015        return this.content;
016    }
017
018    public FileUploadParams setContent(InputStream content) {
019        this.content = content;
020        return this;
021    }
022
023    public String getName() {
024        return this.name;
025    }
026
027    public FileUploadParams setName(String name) {
028        this.name = name;
029        return this;
030    }
031
032    public Date getCreated() {
033        return this.created;
034    }
035
036    public FileUploadParams setCreated(Date created) {
037        this.created = created;
038        return this;
039    }
040
041    public Date getModified() {
042        return this.modified;
043    }
044
045    public FileUploadParams setModified(Date modified) {
046        this.modified = modified;
047        return this;
048    }
049
050    public long getSize() {
051        return this.size;
052    }
053
054    public FileUploadParams setSize(long size) {
055        this.size = size;
056        return this;
057    }
058
059    public ProgressListener getProgressListener() {
060        return this.listener;
061    }
062
063    public FileUploadParams setProgressListener(ProgressListener listener) {
064        this.listener = listener;
065        return this;
066    }
067}
