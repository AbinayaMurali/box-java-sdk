001package com.box.sdk;

003import java.text.ParseException;

012/**
013 * The abstract base class for items in a user's file tree (files, folders, etc.).
014 */
015public abstract class BoxItem extends BoxResource {
016    /**
017     * Constructs a BoxItem for an item with a given ID.
018     * @param  api the API connection to be used by the item.
019     * @param  id  the ID of the item.
020     */
021    public BoxItem(BoxAPIConnection api, String id) {
022        super(api, id);
023    }
024
025    /**
026     * Copies this item to another folder.
027     * @param  destination the destination folder.
028     * @return             info about the copied item.
029     */
030    public abstract BoxItem.Info copy(BoxFolder destination);
031
032    /**
033     * Copies this item to another folder and gives it a new name. If the destination is the same folder as the item's
034     * current parent, then newName must be a new, unique name.
035     * @param  destination the destination folder.
036     * @param  newName     a new name for the copied item.
037     * @return             info about the copied item.
038     */
039    public abstract BoxItem.Info copy(BoxFolder destination, String newName);
040
041    /**
042     * Gets additional information about this item.
043     * @return info about this item.
044     */
045    public abstract BoxItem.Info getInfo();
046
047    /**
048     * Gets additional information about this item that's limited to a list of specified fields.
049     * @param  fields the fields to retrieve.
050     * @return        info about this item containing only the specified fields.
051     */
052    public abstract BoxItem.Info getInfo(String... fields);
053
054    /**
055     * Contains additional information about a BoxItem.
056     */
057    public abstract class Info extends BoxResource.Info {
058        private String sequenceID;
059        private String etag;
060        private String name;
061        private Date createdAt;
062        private Date modifiedAt;
063        private String description;
064        private long size;
065        private List<BoxFolder> pathCollection;
066        private BoxUser.Info createdBy;
067        private BoxUser.Info modifiedBy;
068        private Date trashedAt;
069        private Date purgedAt;
070        private Date contentCreatedAt;
071        private Date contentModifiedAt;
072        private BoxUser.Info ownedBy;
073        private BoxSharedLink sharedLink;
074        private List<String> tags;
075        private BoxFolder.Info parent;
076
077        /**
078         * Constructs an empty Info object.
079         */
080        public Info() {
081            super();
082        }
083
084        /**
085         * Constructs an Info object by parsing information from a JSON string.
086         * @param  json the JSON string to parse.

089            super(json);
090        }
091
092        /**
093         * Constructs an Info object using an already parsed JSON object.
094         * @param  jsonObject the parsed JSON object.
095         */
096        protected Info(JsonObject jsonObject) {
097            super(jsonObject);
098        }
099
100        /**

150         * Sets the description of the item.

154            this.description = description;

272            return this.parent;

275        @Override
276        protected void parseJSONMember(JsonObject.Member member) {
277            super.parseJSONMember(member);
278
279            try {
280                JsonValue value = member.getValue();
281                switch (member.getName()) {
282                    case "sequence_id":
283                        this.sequenceID = value.asString();
284                        break;
285                    case "etag":
286                        this.etag = value.asString();
287                        break;
288                    case "name":
289                        this.name = value.asString();
290                        break;
291                    case "created_at":
292                        this.createdAt = BoxDateParser.parse(value.asString());
293                        break;
294                    case "modified_at":
295                        this.modifiedAt = BoxDateParser.parse(value.asString());
296                        break;
297                    case "description":
298                        this.description = value.asString();
299                        break;
300                    case "size":
301                        this.size = Double.valueOf(value.toString()).longValue();
302                        break;
303                    case "trashed_at":
304                        this.trashedAt = BoxDateParser.parse(value.asString());
305                        break;
306                    case "purged_at":
307                        this.purgedAt = BoxDateParser.parse(value.asString());
308                        break;
309                    case "content_created_at":
310                        this.contentCreatedAt = BoxDateParser.parse(value.asString());
311                        break;
312                    case "content_modified_at":
313                        this.contentModifiedAt = BoxDateParser.parse(value.asString());
314                        break;
315                    case "path_collection":
316                        this.pathCollection = this.parsePathCollection(value.asObject());
317                        break;
318                    case "created_by":
319                        this.createdBy = this.parseUserInfo(value.asObject());
320                        break;
321                    case "modified_by":
322                        this.modifiedBy = this.parseUserInfo(value.asObject());
323                        break;
324                    case "owned_by":
325                        this.ownedBy = this.parseUserInfo(value.asObject());
326                        break;
327                    case "shared_link":
328                        this.sharedLink = new BoxSharedLink(value.asObject());
329                        break;
330                    case "tags":
331                        this.tags = this.parseTags(value.asArray());
332                        break;
333                    case "parent":
334                        JsonObject jsonObject = value.asObject();
335                        String id = jsonObject.get("id").asString();
336                        BoxFolder parentFolder = new BoxFolder(getAPI(), id);
337                        this.parent = parentFolder.new Info(jsonObject);
338                        break;
339                    default:
340                        break;
341                }
342            } catch (ParseException e) {
343                assert false : "A ParseException indicates a bug in the SDK.";
344            }
345        }
346
347        private List<BoxFolder> parsePathCollection(JsonObject jsonObject) {
348            int count = jsonObject.get("total_count").asInt();
349            List<BoxFolder> pathCollection = new ArrayList<BoxFolder>(count);
350            JsonArray entries = jsonObject.get("entries").asArray();
351            for (JsonValue value : entries) {
352                JsonObject entry = value.asObject();
353                String id = entry.get("id").asString();
354                pathCollection.add(new BoxFolder(getAPI(), id));
355            }
356
357            return pathCollection;
358        }
359
360        private BoxUser.Info parseUserInfo(JsonObject jsonObject) {
361            String userID = jsonObject.get("id").asString();
362            BoxUser user = new BoxUser(getAPI(), userID);
363            return user.new Info(jsonObject);
364        }
365
366        private List<String> parseTags(JsonArray jsonArray) {
367            List<String> tags = new ArrayList<String>(jsonArray.size());
368            for (JsonValue value : jsonArray) {
369                tags.add(value.asString());
370            }
371
372            return tags;
373        }
374    }
