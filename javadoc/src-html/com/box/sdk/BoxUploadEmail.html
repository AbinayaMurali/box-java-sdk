001package com.box.sdk;
002
003import com.eclipsesource.json.JsonObject;
004import com.eclipsesource.json.JsonValue;
005
006public class BoxUploadEmail extends BoxJSONObject {
007    private Access access;
008    private String email;
009
010    public BoxUploadEmail() { }
011
012    public BoxUploadEmail(String json) {
013        super(json);
014    }
015
016    BoxUploadEmail(JsonObject jsonObject) {
017        super(jsonObject);
018    }
019
020    public Access getAccess() {
021        return this.access;
022    }
023
024    public void setAccess(Access access) {
025        this.access = access;
026        this.addPendingChange("access", access.toJSONValue());
027    }
028
029    public String getEmail() {
030        return this.email;
031    }
032
033    void parseJSONMember(JsonObject.Member member) {
034        JsonValue value = member.getValue();
035        switch (member.getName()) {
036            case "access":
037                this.access = Access.fromJSONValue(value.asString());
038                break;
039            case "email":
040                this.email = value.asString();
041                break;
042            default:
043                break;
044        }
045    }
046
047    public enum Access {
048        OPEN("open"),
049        COLLABORATORS("collaborators");
050
051        private final String jsonValue;
052
053        private Access(String jsonValue) {
054            this.jsonValue = jsonValue;
055        }
056
057        public static Access fromJSONValue(String jsonValue) {
058            return Access.valueOf(jsonValue.toUpperCase());
059        }
060
061        public String toJSONValue() {
062            return this.jsonValue;
063        }
064    }
065}
